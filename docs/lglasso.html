Identify networks from correlated data (1)

#library(lglasso)
 library(devtools)
 load_all()
# number of nodes
p=20
# number of edge in general network
m1=20
# the difference between the number of edges in individual networks and general network
m2=10
#  number of tissues
m3=4
# number of subjects
n=20
# correlation between tissues
rho=0.8

Model 1: clustered data with no correlation structure

set.seed(5)
cc=matrix(rho,ncol = m3,nrow = m3)
diag(cc)=1
dd=Simulate(type="general",n=n,p=p,m1=m1,m2=m2,m3=m3,cc=cc)
ddata=dd$data

estimate one network

aa=lglasso(data=ddata,lambda = 0.01,type="general",tol = 0.01, trace=TRUE)

## precision matrix
aa$wi

## correlation matrix between different tissues

cov2cor(aa$v[[1]])

estimate two networks based on a grouping variable

group=ifelse(ddata[,2]%in%c("tissue1","tissue2"),1,2)
aa=lglasso(data=ddata,lambda = c(0.1,0.1),type="general",group = group, ,tol = 0.01, trace=TRUE)

## precision matrices
aa$wi

## correlation matrices between tissue 1 and tissue 2
cov2cor(aa$v[[1]])

## correlation matrices between tissue 3 and tissue 4
cov2cor(aa$v[[2]])

Model 2: longitudinal data with exponential correlation structure

library(forcats)
set.seed(1)
dd=Simulate(type="longihomo",n=n,p=p,m1=m1,m2=m2,tau=c(2,2),tt=5)
ddata=do.call(rbind,dd$data)
subjects=fct_inorder(ddata$subject)
group=c(rep(0,nrow(ddata)/2),rep(1,nrow(ddata)/2))

estimate one networks

aa=lglasso(data=ddata,lambda = 0.01,type="expFixed",expFix=1,tol = 0.01, trace=TRUE)

## precision matrix 
aa$wi

## correlation parameter
aa$tau
